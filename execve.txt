int main(int argc, char *argv[])
{
 if(argc < 2)
 {
 printf("No array argument provided.\n");
 return 1;
 }
 int arr[30];
 int j = 0, ele;
 char *token;
 // argv[1] contains the entire string of numbers separated by spaces
 char input_str[300];
 strncpy(input_str, argv[1], sizeof(input_str));
 input_str[sizeof(input_str)-1] = '\0'; // safety null terminate
 // Parse numbers from input_str
 token = strtok(input_str, " ");
 while(token != NULL && j < 30)
 {
 printf("%s removed from char array\n", token);
 arr[j] = atoi(token);
 printf("%d putted in int array\n", arr[j]);
 j++;
 token = strtok(NULL, " ");
 }
 printf("value of j is: %d\n", j);
 printf("Array elements received:\n");
 for(int i = 0; i < j; i++)
 printf("%d\n", arr[i]);
 printf("Enter the elements to be search:\n");
 scanf("%d", &ele);
 binsert(arr, j, ele);
 return 0;
}
void binsert(int arr[], int size, int ele)
{
 int l = 0, r = size - 1, mid;
 while (l <= r)
 {
 mid = (l + r) / 2;
 if (arr[mid] == ele)
 {
 printf("Element %d found at %d location\n", ele, mid + 1);
 return;
 }
 else if (arr[mid] < ele)
 l = mid + 1;
 else
 r = mid - 1;
 }
 printf("Element not found\n");
}
